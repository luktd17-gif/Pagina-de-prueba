from pyscript import document def evaluar(event): resultado = document.getElementById("resultado") try: n1 = int(document.getElementById("nota").value) resultado.className = "" if n1 >= 60 and n1 < 90: texto = "Aprobado" resultado.classList.add("aprobado") elif n1 >= 90 and n1 <= 100: texto = "Excelente" resultado.classList.add("excelente") elif n1 >= 40 and n1 <= 59: texto = "Recuperación" resultado.classList.add("recuperacion") elif n1 < 40: texto = "Reprobado" resultado.classList.add("reprobado") else: texto = "Error: nota no válida" resultado.classList.add("error") resultado.innerText = texto # animación simple sin bloquear resultado.classList.add("animar") except: resultado.innerText = "Ingrese un número válido" resultado.className = "error"
